# .env Environment file

This is the core of the configuration files. It is unique to your system and should **NEVER** be commited to
version control, ie *git*.
```ini
APP_NAME="Banking"
APP_ENV=local
APP_KEY=ABC123
APP_DEBUG=true
APP_URL=https://your-bank-url

LOG_CHANNEL=stack
LOG_LEVEL=debug

DB_CONNECTION=mysql
DB_HOST=127.0.0.1
DB_PORT=3306
DB_DATABASE=bank
DB_USERNAME=
DB_PASSWORD=

BROADCAST_DRIVER=log
CACHE_DRIVER=file
QUEUE_CONNECTION=sync
SESSION_DRIVER=file
SESSION_LIFETIME=120

MEMCACHED_HOST=memcached

REDIS_HOST=redis
REDIS_PASSWORD=null
REDIS_PORT=6379

MAIL_MAILER=smtp
MAIL_HOST=localhost
MAIL_PORT=1025
MAIL_USERNAME=null
MAIL_PASSWORD=null
MAIL_ENCRYPTION=null
MAIL_ADMIN_ADDRESS=me@my-host

AWS_ACCESS_KEY_ID=
AWS_SECRET_ACCESS_KEY=
AWS_DEFAULT_REGION=eu-west-2
AWS_BUCKET=

PUSHER_APP_ID=
PUSHER_APP_KEY=
PUSHER_APP_SECRET=
PUSHER_APP_CLUSTER=mt1

MIX_PUSHER_APP_KEY="${PUSHER_APP_KEY}"
MIX_PUSHER_APP_CLUSTER="${PUSHER_APP_CLUSTER}"

FTP_USERNAME=
FTP_PASSWORD=
FTP_HOST=127.0.0.1
FTP_ROOT=

DEFAULT_USER_PASSWORD=

SCOUT_DRIVER=meilisearch
MEILISEARCH_HOST=http://meilisearch:7700

INFLUXDB_TOKEN=
INFLUXDB_URL=http://127.0.0.1:8086
INFLUX_DB_ORG=Bank
INFLUXDB_BUCKET=transactions
```
